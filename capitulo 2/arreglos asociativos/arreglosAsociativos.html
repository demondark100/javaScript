arreglos asociativo

los arreglos asociativos son un objeto.

pseudocodigo:
variable = {
nombre: contenido,
nombre: contenido,
nombre: contenido
}
console.log(objeto.nombre);
console.log(objeto["nombre"])
javaScript:
let personas = {
nombre: "Pepe",
apellido: "Gonzales",
correo: "pepe@gmail.com"
}
console.log(personas.nombre);
resultado: "Pepe".
console.log(personas["apellido"]);
resultado: "Gonzales"

pseudocodigo:

Object.defineProperty( objeto, "propiedad" , { valor: "valor" })

javaScript:

const objeto = {};
Object.defineProperty(objeto, "color" , {value: "verde"});

esto sera para crear propiedades nuevas a un objeto.

const objeto = {}; Object.defineProperty(objeto, "color" , {value: "verde", writable: false}); console.log(objeto)

con "writable: false" estamos diciendo que esa propiedad no se puede reescribir.

const obj = {
juego: "left 4 dead 2"
}
Object.preventExtensions(obj)
obj.desarrollador = "valve"

con "preventExtensions()" evitaremos que se puedan agregar mas propiedades a un objeto.

const objeto = {
musica: "adventure",
artista: "Coldplay"
}
delete objeto.artista;
console.log(objeto)

con "delete" eliminaremos la propiedad que queramos posdata solo funciona en objetos.
