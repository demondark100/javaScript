parametro rest

los parametros rest son parametros que son indefinidos los cuales se convertiran en arreglos.

ejemplo:

const funcionIndefinida = (...rest) =>{
console.log(rest)
}
funcionIndefinida(5,8,3,"mas parametros")
function masParametros(...rest){
console.log(rest)
}
masParametros(5,8,12,25,"muchos parametros indefinidos")

aqui un ejemplo de como nos puede servir los parametros rest.

const numeros = (...num) =>{
let resultado = 0;
for (let i = 0; i < num.length; i++){
resultado += num[i]
}
console.log(resultado)
}
numeros(5,25)
//esta funcion sumara todos los parametros que pongamos puedes poner los parametros que quieras en esta funcion.

los parametros rest siempre deben ser el ultimo parametro de todos.

ejemplo:

error

function error_1(...rst,a){
console.log(`${rst} ${a}`)
}
error_1(8,10)

const error_2 =(...rst,a) =>{
console.log(`${rst} ${a}`)
}
error_1(8,10)

correcto

function correcto_1(a,...rst){
console.log(`${a} - ${rst}`)
}
correcto_1(8,10)

const correcto_2 = (a,...rst) =>{
console.log(`${a} - ${rst}`)
}
correcto_2(8,10)

esto pasa proque los parametros rest son ilimitados/indefinidos por eso siempre deben ir al final.
