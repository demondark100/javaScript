caracteristicas de prototipos

cadena

let cadena = "hola mundo";
console.log(cadena.__proto__)

arreglo

let arreglo = [
"number",
55,
false
];
console.log(arreglo)

la unica propiedad de el arreglo es "length", las demas propiedades es lo que eredara del prototipo.

objeto

class Objeto {
constructor(){}
saludar(){
console.log("hola")
}
}
const objeto = new Objeto()
console.log(objeto.__proto__.__proto__)

//"__proto__" es la erencia del "prototype" solamente que lo estamos modificando.

diferencias entre sobreescribir proto y sobreescribir un metodo.

class Objeto {
constructor(){}
saludar(){
console.log("hola")
}
}
const objeto = new Objeto()
objeto.saludar = () =>{
console.log("hola desde afuera")
}
objeto.saludar()
// resultado: "hola desde afuera"

esto pasa porque se ejecuta primero la parte de adentro y luego los metodos de afuera , no estamos modificando el prototipo solo estamos agregando.

class Objeto {
constructor(){}
saludar(){
console.log("hola")
}
}
const objeto = new Objeto;
objeto.__proto__.saludar = ()=>{ console.log("hola desde afuera")
}
console.log(objeto)

//aqui si estamos modificando el prototipo del metodo "saludar".

objeto

class Objeto {
constructor(){}
saludar(){
console.log("hola")
}
}
const objeto = new Objeto;

let arr = [];
arr.__proto__ = objeto
arr.saludar()

//con esto estamos eredando un prototipo al arreglo.
