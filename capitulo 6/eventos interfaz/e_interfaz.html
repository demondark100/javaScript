eventos de la interfaz

HTML:

p class="parrafo"> hola mundo /p>

javaScript:

const parrafo = document.querySelector(".parrafo");
parrafo.addEventListener("error",()=>{
alert("ha sucedido un error")
});

con "error" el evento se activara cuando pase un error por ejemplo no cargo una imagen.

window.addEventListener("load",()=>{
alert("el sitio web a cargado correctamente.")
});

con "load" el evento se activara cuando el elemento que indiquemos haya cargado por completo.

javaScript:

addEventListener("resize",()=>{
alert("se cambio la resolucion")
});

"resize" se ejecutara cuando la resolucion se cambie.

javaScript:

addEventListener("scroll",()=>{
alert("se escroleo el sitio web")
});

"scroll" se ejecutara cuando se escrolee en el sitio web.

HTML:

input class="seleccion" type="text"> div class="mostrar"> /div>

javaScript:

seleccion.addEventListener("select",(e)=>{
console.log(e.target.selectionStart)
console.log(e.target.selectionEnd)
})

con "select" podemos ver que selecciono el usuario de algun input.

con "selectionStart" y "selectionEnd" podremos ver la posision del inicio y final de donde selecciono el usuario.

javaScript:

const seleccion = document.querySelector(".seleccion");
const mostrar = document.querySelector(".mostrar");

seleccion.addEventListener("select",(e)=>{
let posision_1 = e.target.selectionStart;
let posision_2 = e.target.selectionEnd;
let seleccionado = seleccion.value.substring(posision_1,posision_2);
mostrar.textContent = seleccionado
})
