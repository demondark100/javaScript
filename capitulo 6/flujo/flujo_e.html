flujo de eventos

con esto veremos cual es el flujo de los eventos.

HTML:

div class="content_1">
div class="content_2">
button class="boton"> click /button>
/div>
/div>

css:

.content_1{
width: 250px;
height: 110px;
background-color: #0f0;
}
.content_2{
background-color: #f00;
width: 150px;
height: 85px;
margin: auto;
}
.boton{
background-color: #00f;
color: #fff;
}

javaScript:

const content_1 = document.querySelector(".content_1");
const content_2 = document.querySelector(".content_2");
const boton = document.querySelector(".boton");
content_1.addEventListener("click",()=>{
alert("content_1");
});
content_2.addEventListener("click",()=>{
alert("content_2");
});
boton.addEventListener("click",()=>{
alert("click");
});

javaScript:

const content_1 = document.querySelector(".content_1");
const content_2 = document.querySelector(".content_2");
const boton = document.querySelector(".boton");
content_1.addEventListener("click",()=>{
alert("content_1");
});
content_2.addEventListener("click",()=>{
alert("content_2");
},true);
boton.addEventListener("click",()=>{
alert("click");
});

con "true" manejaremos el flujo de los eventos en este caso "content_2" se ejecutara primero

para ver resultados hacer click en "click"

javaScript:

const content_1 = document.querySelector(".content_1");
const content_2 = document.querySelector(".content_2");
const boton = document.querySelector(".boton");
content_1.addEventListener("click",()=>{
alert("content_1");
});
content_2.addEventListener("click",()=>{
alert("content_2");
});
boton.addEventListener("click",(e)=>{
alert("click");
e.stopPropagation();
});

con "stopPropagation()" vamos a detener la propagacion de los eventos.

para ver resultados hacer click en "click"
